45  class Queue {
46  public:
47  bool push(T const & value) {
48  std::unique_lock<std::mutex> lock(this->mutex);
49  this->q.push(value);
50  return true;
51  }
52  // deletes the retrieved element, do not use for non integral types
53  bool pop(T & v) {
54  std::unique_lock<std::mutex> lock(this->mutex);
55  if (this->q.empty())
56  return false;
57  v = this->q.front();
58  this->q.pop();
59  return true;
60  }
61  bool empty() {
62  std::unique_lock<std::mutex> lock(this->mutex);
63  return this->q.empty();
64  }

91  // change the number of threads in the pool
92  // should be called from one thread, otherwise be careful to not interleave, also with this->stop()
93  // nThreads must be >= 0
94  void resize(int nThreads) {
95  if (!this->isStop && !this->isDone) {
96  int oldNThreads = static_cast<int>(this->threads.size());
97  if (oldNThreads <= nThreads) { // if the number of threads is increased
98  this->threads.resize(nThreads);
99  this->flags.resize(nThreads);
100 
101  for (int i = oldNThreads; i < nThreads; ++i) {
102  this->flags[i] = std::make_shared<std::atomic<bool>>(false);
103  this->set_thread(i);
104  }
105  }
106  else { // the number of threads is decreased
107  for (int i = oldNThreads - 1; i >= nThreads; --i) {
108  *this->flags[i] = true; // this thread will finish
109  this->threads[i]->detach();
110  }
111  {
112  // stop the detached threads that were waiting
113  std::unique_lock<std::mutex> lock(this->mutex);
114  this->cv.notify_all();
115  }
116  this->threads.resize(nThreads); // safe to delete because the threads are detached
117  this->flags.resize(nThreads); // safe to delete because the threads have copies of shared_ptr of the flags, not originals
118  }
119  }
120  }

140  // wait for all computing threads to finish and stop all threads
141  // may be called asynchronously to not pause the calling thread while waiting
142  // if isWait == true, all the functions in the queue are run, otherwise the queue is cleared without running the functions
143  void stop(bool isWait = false) {
144  if (!isWait) {
145  if (this->isStop)
146  return;
147  this->isStop = true;
148  for (int i = 0, n = this->size(); i < n; ++i) {
149  *this->flags[i] = true; // command the threads to stop
150  }
151  this->clear_queue(); // empty the queue
152  }
153  else {
154  if (this->isDone || this->isStop)
155  return;
156  this->isDone = true; // give the waiting threads a command to finish
157  }
158  {
159  std::unique_lock<std::mutex> lock(this->mutex);
160  this->cv.notify_all(); // stop all waiting threads
161  }
162  for (int i = 0; i < static_cast<int>(this->threads.size()); ++i) { // wait for the computing threads to finish
163  if (this->threads[i]->joinable())
164  this->threads[i]->join();
165  }
166  // if there were no threads in the pool but some functors in the queue, the functors are not deleted by the threads
167  // therefore delete them here
168  this->clear_queue();
169  this->threads.clear();
170  this->flags.clear();
171  }

174  auto push(F && f, Rest&&... rest) ->std::future<decltype(f(0, rest...))> {
175  auto pck = std::make_shared<std::packaged_task<decltype(f(0, rest...))(int)>>(
176  std::bind(std::forward<F>(f), std::placeholders::_1, std::forward<Rest>(rest)...)
177  );
178  auto _f = new std::function<void(int id)>([pck](int id) {
179  (*pck)(id);
180  });
181  this->q.push(_f);
182  std::unique_lock<std::mutex> lock(this->mutex);
183  this->cv.notify_one();
184  return pck->get_future();
185  }

210  void set_thread(int i) {
211  std::shared_ptr<std::atomic<bool>> flag(this->flags[i]); // a copy of the shared ptr to the flag
212  auto f = [this, i, flag/* a copy of the shared ptr to the flag */]() {
213  std::atomic<bool> & _flag = *flag;
214  std::function<void(int id)> * _f;
215  bool isPop = this->q.pop(_f);
216  while (true) {
217  while (isPop) { // if there is anything in the queue
218  std::unique_ptr<std::function<void(int id)>> func(_f); // at return, delete the function even if an exception occurred
219  (*_f)(i);
220  if (_flag)
221  return; // the thread is wanted to stop, return even if the queue is not empty yet
222  else
223  isPop = this->q.pop(_f);
224  }
225  // the queue is empty here, wait for the next command
226  std::unique_lock<std::mutex> lock(this->mutex);
227  ++this->nWaiting;
228  this->cv.wait(lock, [this, &_f, &isPop, &_flag](){ isPop = this->q.pop(_f); return isPop || this->isDone || _flag; });
229  --this->nWaiting;
230  if (!isPop)
231  return; // if the queue is empty and this->isDone == true or *flag then return
232  }
233  };
234  this->threads[i].reset(new T(f)); // compiler may not support std::make_unique()
235  }
