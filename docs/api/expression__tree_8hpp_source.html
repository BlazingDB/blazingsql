170 struct custom_op_transformer : public node_transformer {
171 public:
172  node * transform(operad_node& node) override { return &node; }
173 
174  node * transform(operator_node& node) override {
175  if(node.value == "CASE") {
176  return transform_case(node, 0);
177  } else if(node.value == "Reinterpret") {
178  return remove_reinterpret(node);
179  } else if(node.value == "ROUND") {
180  return transform_round(node);
181  } else if(StringUtil::beginsWith(node.value, "CAST")) {
182  return transform_cast_literal(node);
183  }
184 
185  return &node;
186  }
187 
188 private:
189  node * transform_case(operator_node& case_node, size_t child_idx) {
190  assert(case_node.children.size() >= 3 && case_node.children.size() % 2 != 0);
191  assert(child_idx < case_node.children.size());
192 
193  if (child_idx == case_node.children.size() - 1) {
194  return case_node.children[child_idx].release();
195  }
196 
197  auto condition = std::move(case_node.children[child_idx]);
198  auto then = std::move(case_node.children[child_idx + 1]);
199 
200  auto magic_if_not = std::unique_ptr<node>(new operator_node("MAGIC_IF_NOT"));
201  magic_if_not->children.push_back(std::move(condition));
202  magic_if_not->children.push_back(std::move(then));
203 
204  node * first_non_magic = new operator_node{"FIRST_NON_MAGIC"};
205  first_non_magic->children.push_back(std::move(magic_if_not));
206  first_non_magic->children.push_back(std::unique_ptr<node>(transform_case(case_node, child_idx + 2)));
207 
208  return first_non_magic;
209  }
