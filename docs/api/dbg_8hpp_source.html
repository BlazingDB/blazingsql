62 template <typename T>
63 std::string type_name() {
64 #if defined(__clang__)
65  size_t prefixlen =
66  sizeof("const char *dbg_macro::type_name_impl() [T = ") - 1;
67  size_t suffixlen = sizeof("]") - 1;
68 #elif defined(__GNUC__) && !defined(__clang__)
69  size_t prefixlen =
70  sizeof("const char* dbg_macro::type_name_impl() [with T = ") - 1;
71  size_t suffixlen = sizeof("]") - 1;
72 #elif defined(_MSC_VER)
73  size_t prefixlen =
74  sizeof("const char *__cdecl dbg_macro::type_name_impl<") - 1;
75  size_t suffixlen = sizeof(">(void)") - 1;
76 #else
77 #error "No support for this compiler."
78 #endif
79  std::string type = type_name_impl<T>();
80  return type.substr(prefixlen, type.size() - prefixlen - suffixlen);

196 template <typename Container>
197 typename std::enable_if<has_begin_end_size<Container>::value, bool>::type
198 prettyPrint(std::ostream& stream, Container const& value) {
199  stream << "{";
200  const size_t size = value.size();
201  const size_t n = std::min(size_t{5}, size);
202  size_t i = 0;
203  for (auto it = begin(value); it != end(value) && i < n; ++it, ++i) {
204  prettyPrint(stream, *it);
205  if (i != n - 1) {
206  stream << ", ";
207  }
208  }
209 
210  if (size > n) {
211  stream << ", ...";
212  stream << " size:" << size;
213  }
214 
215  stream << "}";
216  return true;
217 }

241  template <typename T>
242  T&& print(T&& value) const {
243  const T& ref = value;
244  std::stringstream stream_value;
245  const bool print_expr_and_type = prettyPrint(stream_value, ref);
246 
247  std::cerr << ansi(ANSI_DEBUG) << "[" << m_filepath << ":" << m_line << " ("
248  << m_function_name << ")] " << ansi(ANSI_RESET);
249  if (print_expr_and_type) {
250  std::cerr << ansi(ANSI_EXPRESSION) << m_expression << ansi(ANSI_RESET)
251  << " = ";
252  }
253  std::cerr << ansi(ANSI_VALUE) << stream_value.str() << ansi(ANSI_RESET);
254  if (print_expr_and_type) {
255  using ValueType = typename std::remove_cv<
256  typename std::remove_reference<T>::type>::type;
257  std::cerr << " (" << ansi(ANSI_TYPE) << type_name<ValueType>()
258  << ansi(ANSI_RESET) << ")";
259  }
260  std::cerr << std::endl;
261 
262  return std::forward<T>(value);
263  }
